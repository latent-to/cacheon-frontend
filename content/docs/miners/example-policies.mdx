## INT8 quantization

Quantize K and V to INT8 with per-row symmetric scales on write, dequantize back to full precision on attend.

```python
class Int8Policy(KVCachePolicy):
    def setup(self, config):
        self.k_cache = [[] for _ in range(config.num_layers)]
        self.v_cache = [[] for _ in range(config.num_layers)]
        self.k_scales = [[] for _ in range(config.num_layers)]
        self.v_scales = [[] for _ in range(config.num_layers)]

    def write(self, keys, values, layer_idx, positions):
        k_scale = keys.abs().amax(dim=-1, keepdim=True) / 127
        v_scale = values.abs().amax(dim=-1, keepdim=True) / 127
        k_int8 = (keys / k_scale.clamp(min=1e-8)).round().to(torch.int8)
        v_int8 = (values / v_scale.clamp(min=1e-8)).round().to(torch.int8)
        self.k_cache[layer_idx].append(k_int8)
        self.v_cache[layer_idx].append(v_int8)
        self.k_scales[layer_idx].append(k_scale)
        self.v_scales[layer_idx].append(v_scale)

    def attend(self, query, layer_idx):
        K_int = torch.cat(self.k_cache[layer_idx], dim=2).float()
        V_int = torch.cat(self.v_cache[layer_idx], dim=2).float()
        K_scale = torch.cat(self.k_scales[layer_idx], dim=2)
        V_scale = torch.cat(self.v_scales[layer_idx], dim=2)
        K = (K_int * K_scale).to(query.dtype)
        V = (V_int * V_scale).to(query.dtype)
        # GQA broadcast
        K = K.repeat_interleave(query.shape[1] // K.shape[1], dim=1)
        V = V.repeat_interleave(query.shape[1] // V.shape[1], dim=1)
        scores = query @ K.transpose(-1, -2) / (query.shape[-1] ** 0.5)
        weights = scores.softmax(dim=-1)
        return AttentionOutput(output=weights @ V, attention_weights=weights)
```

**Limitation**: Decompresses K/V back to full precision during `attend`, so the full-size buffer exists in memory at scoring time. The storage savings don't fully translate to peak memory savings.

## H2O eviction

Tracks per-token attention mass over time and evicts the lowest-importance tokens when the cache exceeds a fixed budget.

```python
class H2OPolicy(KVCachePolicy):
    def __init__(self, heavy_budget=512, recent_budget=512):
        self.heavy_budget = heavy_budget
        self.recent_budget = recent_budget

    def setup(self, config):
        self.k = [None] * config.num_layers
        self.v = [None] * config.num_layers
        self.importance = [None] * config.num_layers
        self.positions = [None] * config.num_layers

    def write(self, keys, values, layer_idx, positions):
        if self.k[layer_idx] is None:
            self.k[layer_idx] = keys
            self.v[layer_idx] = values
            self.importance[layer_idx] = torch.zeros(
                keys.shape[:-1], device=keys.device, dtype=torch.float32
            )
            self.positions[layer_idx] = positions
        else:
            self.k[layer_idx] = torch.cat([self.k[layer_idx], keys], dim=2)
            self.v[layer_idx] = torch.cat([self.v[layer_idx], values], dim=2)
            new_imp = torch.zeros(keys.shape[:-1], device=keys.device, dtype=torch.float32)
            self.importance[layer_idx] = torch.cat([self.importance[layer_idx], new_imp], dim=-1)
            self.positions[layer_idx] = torch.cat([self.positions[layer_idx], positions], dim=-1)

        self._evict(layer_idx)

    def _evict(self, layer_idx):
        total = self.heavy_budget + self.recent_budget
        if self.k[layer_idx].shape[2] <= total:
            return
        # Keep last `recent_budget` tokens unconditionally; among the rest,
        # keep top-`heavy_budget` by importance.
        keep_recent = self.k[layer_idx][..., -self.recent_budget:, :]
        # ... (see H2O paper for full indexing logic)

    def attend(self, query, layer_idx):
        K = self.k[layer_idx].repeat_interleave(query.shape[1] // self.k[layer_idx].shape[1], dim=1)
        V = self.v[layer_idx].repeat_interleave(query.shape[1] // self.v[layer_idx].shape[1], dim=1)
        scores = query @ K.transpose(-1, -2) / (query.shape[-1] ** 0.5)
        weights = scores.softmax(dim=-1)
        # Update importance with this step's attention mass
        self.importance[layer_idx] += weights.sum(dim=(0, 2)).to(self.importance[layer_idx].dtype)
        return AttentionOutput(output=weights @ V, attention_weights=weights)
```

**Limitation**: Aggressive eviction budgets risk failing the KL gate on reasoning-heavy prompts where dropped tokens turn out to matter. The `_evict` logic here is incomplete; the full indexing is left as an exercise.

## TurboQuant (fused scoring sketch)

Rotates K into a random orthonormal basis, quantizes to 3 bits with Lloyd-Max centroids, and scores directly against the compressed representation using a QJL correction term. K is not expanded back to full precision in this sketch.

```python
class TurboQuantPolicy(KVCachePolicy):
    BITS = 3
    N_CENTROIDS = 8  # 2 ** BITS

    def setup(self, config):
        self.config = config
        # Random rotation matrix per layer, fixed for the whole sequence.
        torch.manual_seed(0xC0FFEE)  # deterministic for reproducibility
        self.R = [
            torch.linalg.qr(torch.randn(config.head_dim, config.head_dim, device="cuda"))[0]
            for _ in range(config.num_layers)
        ]
        # Lloyd-Max centroids pre-computed for a standard normal.
        self.centroids = self._compute_centroids()

        self.k_quant = [[] for _ in range(config.num_layers)]
        self.v_cache = [[] for _ in range(config.num_layers)]
        self.qjl_signs = [[] for _ in range(config.num_layers)]

    def write(self, keys, values, layer_idx, positions):
        R = self.R[layer_idx]
        k_rot = keys @ R                                   # rotated keys
        k_3bit = self._quantize_lloyd_max(k_rot)           # [..., head_dim], values in [0, 7]
        k_signs = (k_rot > 0).to(torch.uint8)              # QJL sign bits for correction

        self.k_quant[layer_idx].append(k_3bit)
        self.v_cache[layer_idx].append(values)             # V stays FP16 in this sketch
        self.qjl_signs[layer_idx].append(k_signs)

    def attend(self, query, layer_idx):
        R = self.R[layer_idx]
        q_rot = query @ R
        K_3bit = torch.cat(self.k_quant[layer_idx], dim=2)
        K_signs = torch.cat(self.qjl_signs[layer_idx], dim=2)
        V = torch.cat(self.v_cache[layer_idx], dim=2)

        # Score directly against centroid values, without expanding to FP16.
        scores = self._fused_score(q_rot, K_3bit, K_signs)
        weights = scores.softmax(dim=-1)
        # GQA broadcast V for the aggregate
        V = V.repeat_interleave(q_rot.shape[1] // V.shape[1], dim=1)
        return AttentionOutput(output=weights @ V, attention_weights=None)

    def _compute_centroids(self):
        # Lloyd-Max optimal 8-level quantization of N(0, 1)
        return torch.tensor([-2.152, -1.344, -0.756, -0.245, 0.245, 0.756, 1.344, 2.152])
```
